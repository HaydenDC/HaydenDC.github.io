Local Storage



function countUp()
	{
		//Check to see if browser supports local storage
		if(typeof(Storage) !== "undefined")
		{
			//Make local storage object
			if(localStorage.clickCnt)
			{
				//Converts storage from a string to a number
				localStorage.clickCnt = Number(localStorage.clickCnt)+1;
			}

			//If clickCnt doesn't exist
			else
			{
				localStorage.clickCnt = 1;
			}
			document.getElementById("counter").innerHTML = "Count is: " + localStorage.clickCnt;
		}

		//If there is no storage..
		else
		{
			document.getElementById("counter").innerHTML = "This browser does not support web storage. Sorry";
		}
	}
