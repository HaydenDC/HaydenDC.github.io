Session Storage



function countUp()
	{
		//Check to see if browser supports session storage
		if(typeof(Storage) !== "undefined")
		{
			//Make session storage object
			if(sessionStorage.clickCnt)
			{
				//Converts storage from a string to a number
				sessionStorage.clickCnt = Number(sessionStorage.clickCnt)+1;
			}

			//If clickCnt doesn't exist
			else
			{
				sessionStorage.clickCnt = 1;
			}
			document.getElementById("counter").innerHTML = "Count is: " + sessionStorage.clickCnt;
		}

		//If there is no storage..
		else
		{
			document.getElementById("counter").innerHTML = "This browser does not support web storage. Sorry";
		}
	}
